        // Initialize Lucide icons

        // --- p5.js Sketch ---
        const sketch = (p) => {
            let particles = [];
            const numParticles = 50;
            const maxDistance = 100;

            p.setup = () => {
                const canvas = p.createCanvas(p.windowWidth, p.windowHeight);
                canvas.parent('p5-canvas-container'); // Attach to the specific div
                canvas.style('position', 'absolute');
                canvas.style('top', '0');
                canvas.style('left', '0');
                canvas.style('z-index', '-1');
                canvas.style('opacity', '0.05');
                canvas.style('pointer-events', 'none');

                for (let i = 0; i < numParticles; i++) {
                    particles.push(new Particle(p));
                }
            };

            p.draw = () => {
                p.clear();
                p.stroke(200, 200, 200, 30);
                p.strokeWeight(0.5);

                for (let i = 0; i < particles.length; i++) {
                    particles[i].update();
                    particles[i].display();

                    for (let j = i + 1; j < particles.length; j++) {
                        const d = p.dist(particles[i].pos.x, particles[i].pos.y, particles[j].pos.x, particles[j].pos.y);
                        if (d < maxDistance) {
                            p.line(particles[i].pos.x, particles[i].pos.y, particles[j].pos.x, particles[j].pos.y);
                        }
                    }
                }
            };

            p.windowResized = () => {
                p.resizeCanvas(p.windowWidth, p.windowHeight);
                particles = [];
                for (let i = 0; i < numParticles; i++) {
                    particles.push(new Particle(p));
                }
            };
        };

        class Particle {
            constructor(p) {
                this.p = p;
                this.pos = p.createVector(p.random(p.width), p.random(p.height));
                this.vel = p.createVector(p.random(-0.5, 0.5), p.random(-0.5, 0.5));
                this.size = p.random(1, 3);
                this.color = p.color(220, 220, 220, 50);
            }

            update() {
                this.pos.add(this.vel);
                if (this.pos.x < 0) this.pos.x = this.p.width;
                if (this.pos.x > this.p.width) this.pos.x = 0;
                if (this.pos.y < 0) this.pos.y = this.p.height;
                if (this.pos.y > this.p.height) this.pos.y = 0;
            }

            display() {
                this.p.noStroke();
                this.p.fill(this.color);
                this.p.ellipse(this.pos.x, this.pos.y, this.size);
            }
        }

        function drawSkillsChart() {
            const skillsData = [
                { name: 'Python', proficiency: 90 },
                { name: 'Scikit-learn', proficiency: 85 },
                { name: 'SQL', proficiency: 80 },
                { name: 'Tableau', proficiency: 75 },
                { name: 'C++', proficiency: 70 },
                { name: 'Tensorflow', proficiency: 70 },
                { name: 'NumPy', proficiency: 88 },

                { name: 'Seaborn', proficiency: 78 },
                { name: 'Mediapipe', proficiency: 65 },
                { name: 'Blazepose', proficiency: 60 },
                { name: 'MySQL', proficiency: 82 },
                { name: 'PowerBI', proficiency: 70 },
            ];

            const svgElement = document.getElementById('skills-chart');
            if (!svgElement) return; // Exit if element not found

            const svg = d3.select(svgElement);
            const parentWidth = svgElement.parentElement.clientWidth;
            const margin = { top: 20, right: 20, bottom: 30, left: 100 };
            const width = parentWidth - margin.left - margin.right;
            const height = skillsData.length * 30;

            svg.selectAll('*').remove(); // Clear previous chart elements

            const g = svg.attr("width", width + margin.left + margin.right)
                         .attr("height", height + margin.top + margin.bottom)
                         .append("g")
                         .attr("transform", `translate(${margin.left},${margin.top})`);

            const xScale = d3.scaleLinear()
                .domain([0, 100])
                .range([0, width]);

            const yScale = d3.scaleBand()
                .domain(skillsData.map(d => d.name))
                .range([0, height])
                .padding(0.2);

            g.append("g")
                .attr("class", "x-axis")
                .call(d3.axisTop(xScale).ticks(5))
                .selectAll("text")
                .attr("fill", "#a7b7c5");

            g.append("g")
                .attr("class", "y-axis")
                .call(d3.axisLeft(yScale))
                .selectAll("text")
                .attr("fill", "#e6edf3");

            g.selectAll(".bar")
                .data(skillsData)
                .enter().append("rect")
                .attr("class", "bar")
                .attr("x", 0)
                .attr("y", d => yScale(d.name))
                .attr("width", d => xScale(d.proficiency))
                .attr("height", yScale.bandwidth())
                .attr("fill", "#0e7490")
                .attr("rx", 4)
                .attr("ry", 4)
                .on("mouseover", function(event, d) {
                    d3.select(this).attr("fill", "#22d3ee");
                    g.append("text")
                        .attr("class", "proficiency-text")
                        .attr("x", xScale(d.proficiency) + 5)
                        .attr("y", yScale(d.name) + yScale.bandwidth() / 2 + 5)
                        .attr("fill", "#e6edf3")
                        .attr("font-size", "0.8em")
                        .text(`${d.proficiency}%`);
                })
                .on("mouseout", function() {
                    d3.select(this).attr("fill", "#0e7490");
                    g.selectAll(".proficiency-text").remove();
                });

            g.selectAll(".label")
                .data(skillsData)
                .enter().append("text")
                .attr("class", "label")
                .attr("x", d => xScale(d.proficiency) + 5)
                .attr("y", d => yScale(d.name) + yScale.bandwidth() / 2 + 5)
                .attr("fill", "#a7b7c5")
                .attr("font-size", "0.8em")
                .text(d => `${d.proficiency}%`);
        }

                navButtons.forEach(button => {

                    }
                });

                // Redraw D3 chart if Skills section is active
                if (sectionId === 'section-skills') {
                    drawSkillsChart();
                }
            };

            // Add click listeners to navigation buttons
            navButtons.forEach(button => {
                button.addEventListener('click', () => {
                    const sectionId = `section-${button.id.replace('nav-', '')}`;
                    showSection(sectionId);
                });

            showSection('section-summary');

            // Trigger fade-in for the body
            bodyContainer.classList.remove('opacity-0');
            bodyContainer.classList.add('opacity-100');

            window.addEventListener('resize', () => {
                if (!document.getElementById('section-skills').classList.contains('hidden')) {
                    drawSkillsChart();
                }
            });
